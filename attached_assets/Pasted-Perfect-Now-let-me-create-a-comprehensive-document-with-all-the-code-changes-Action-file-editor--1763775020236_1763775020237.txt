        let cartSubtotal = 0;
        
        function loadCheckout() {
            const cart = JSON.parse(localStorage.getItem(\"cart\")) || [];
            const checkoutItems = document.getElementById(\"checkoutItems\");
            
            if (cart.length === 0) {
                window.location.href = \"page7.html\";

            }
            
            let subtotal = 0;
            checkoutItems.innerHTML = \"\";
            
            cart.forEach(item => {
                const price = parseFloat(item.price.replace(\"$\", \"\"));
                subtotal += price;
                
                const itemDiv = document.createElement(\"div\");
                itemDiv.style.cssText = \"display: flex; justify-content: space-between; padding: 10px 0; border-bottom: 1px solid var(--border-color);\";
                const colorInfo = item.color && item.color !== \"Default\" ? ` (${item.color})` : '';
                itemDiv.innerHTML = `
                    <span>${item.name}${colorInfo}</span>
                    <span>${item.price}</span>
                `;
                checkoutItems.appendChild(itemDiv);
            });
            
            cartSubtotal = subtotal;
            document.getElementById(\"checkoutSubtotal\").textContent = \"$\" + subtotal.toFixed(2);
            updateTotal();
        }
        
        function updateTotal() {

        }

                banner.id = \"notificationBanner\";
                banner.className = \"notification-banner\";
                document.body.appendChild(banner);

            banner.classList.add(\"show\");

                banner.classList.remove(\"show\");
